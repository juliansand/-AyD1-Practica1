public int Potencia(int numero){
  double res;
  res = Math.Pow(numero , 2);
  return res;
}

public int raiz(int numero){
  double res;
  res = Math.Sqrt(numero)
  
  return res;
}

public int PotenciaCubo(int numero){
  double res;
  res = Math.Pow(numero , 3);
  return res;
}

public int Raizcubica(int numero){
  double res;
  res = Math.Pow(numero , 1/3);
  return res;
}
